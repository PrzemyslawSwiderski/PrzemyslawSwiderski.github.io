private fun tryToParse(yamlString: String): MdMetadata {
    try {
        return Yaml.default.decodeFromString<MdMetadata>(yamlString)
    } catch (ex: Exception) {
        println("Exception during YAML parsing: ${ex.message}")
        return MdMetadata()
    }
}
